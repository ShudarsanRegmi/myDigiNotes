

# 01 - Square

```
* * * * 
* * * * 
* * * * 
* * * * 
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 4;
    for (int i = 0; i<n; i++) {
        for (int j = 0; j<n; j++) {
            cout << "* " ;
        }
        cout << endl;
    }
    return 0;
}
```

# 06 - Pyramid

```
       *
      ***
     *****
    *******
   *********
  ***********
 *************
***************
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 7;
    for (int i = 0; i <= n; i++) {
        for (int k = n-i-1; k>=0; k--) {
            cout << " ";
        }
        for (int j = 1; j<=2*i+1; j++) { // j< (1,3,5,7)
            cout << "*";
        }
        cout << endl;
    }
    return 0;
}
```
