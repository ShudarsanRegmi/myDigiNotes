

# 01 - Square

```
* * * * 
* * * * 
* * * * 
* * * * 
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 4;
    for (int i = 0; i<n; i++) {
        for (int j = 0; j<n; j++) {
            cout << "* " ;
        }
        cout << endl;
    }
    return 0;
}
```

# 02 - Hollow Square

```
* * * * * * * 
*           * 
*           * 
*           * 
*           * 
*           * 
* * * * * * * 
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 7;
    for (int i = 0; i<n; i++) {
        for (int j = 0; j<n; j++) {
            if (j == 0 || j == n-1 || i == 0 || i == n-1) {
                cout << "* ";
            }else {
                cout << "  ";
            }
        }
        cout << endl;
    }
    return 0;
}
```

# 03 - Rhombus

```
* * * * * * * 
 * * * * * * * 
  * * * * * * * 
   * * * * * * * 
    * * * * * * * 
     * * * * * * * 
      * * * * * * * 
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 7;
    for (int i = 0; i < n; i++) {
        for (int k = 0; k < i; k++) {
            cout << " ";
        }
        for (int j = 0; j < n; j++) {
            cout << "* ";
        }
        cout << endl;
    }
    return 0;
}
```

# 04 - Mirror Rhombus

```
       * * * * * * * 
      * * * * * * * 
     * * * * * * * 
    * * * * * * * 
   * * * * * * * 
  * * * * * * * 
 * * * * * * * 
```

```cpp
#include <iostream>

using namespace std;


int main() {
    int n = 7;
    for (int i = 0; i < n; i++) {
        for (int k = 0; k < n-i; k++) {
            cout << " ";
        }
        for (int j = 0; j < n; j++) {
            cout << "* ";
        }
        cout << endl;
    }
    return 0;
}
```
